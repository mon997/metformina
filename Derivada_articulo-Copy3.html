def PK_model(y,t,kgg,kgo,kgl,kls,ksl,ksg,kso):       
    A,B,C,D=y  
    
    dA_dt = (-A*kgg+kgo) 
    dB_dt = (A*kgg+D*ksg-B*kgl) 
    dC_dt = (B*kgl+D*ksl-C*kls)
    dD_dt = (C*kls- D*ksl+ksg+kso)
    
    return([dA_dt, dB_dt, dC_dt, dD_dt])  

# Tiempo
t = numpy.linspace(0, 1,60)  

# Resultados  
solution = scipy.integrate.odeint(PK_model, [A0, B0, C0, D0], t, args=(kgg,kgo,kgl,kls,ksl,ksg,kso)) 
solution =numpy.array(solution)

kin= 2.08            #zero-order rate of glucose input into the body
kout= 5.42**(-3)   # first-order rate of glucose utilization

ConI = Dosis_metformina*(kgg+kgo)  #aumento en el uso de glucosa en intestino
EmaxI= 0.486                       #Efecto maximo de metformina intestino
ECon50I= 431                       #Concentracion de metformina que produce 50% del efecto maximo (ug)
shI=2                              #Shape factor

ConH = Dosis_metformina*kls    #aumento en el uso de glucosa en higado
ImaxH= 0.378                   #Efecto maximo de metformina higado
ICon50H= 521                   #Concentracion de metformina que produce 50% del efecto maximo (ug)
shH=5                          #Shape factor

ConP = Dosis_metformina*kso #aumento en el uso de glucosa en periferia
EmaxP= 0.148;               #Efecto maximo de metformina periferia
ECon50P = 1024              #Concentracion de metformina que produce 50% del efecto maximo (ug)
shP=5                       #Shape factor

## Modelo PK-PD (= weight coefficient metformina)ARTICULO 1
WC_Instestino  = (EmaxI*(ConI**(shI)))/(ECon50I**(shI))+ ConI 
WC_Higado = (ImaxH*(ConH**(shH)))/((ICon50H**(shH))+ ConH)
WC_Periferia = (EmaxP*(ConP**(shP)))/((ECon50P**(shP))+ ConP)

## Tasa de consumo de glucosa con metformina ARTICULO 2

# Tasa de consumo de glucosa sin metformina intestino
tglucose_intestino= 0.76 
# Tasa de uso de glucosa con metformina
RPKPD_Intestino=(1+WC_Instestino)*tglucose_intestino

def PKPD_model(y,t,kin,kout, tglucose_higado,RPKPD_Higado, RPKPD_Intestino, RPKPD_Periferia):

    dR_dt = kin*(tglucose_higado-RPKPD_Higado)- kout*(1+RPKPD_Intestino+RPKPD_Periferia)*R

    return(dR_dt)
